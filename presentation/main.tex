\begin{frame}{Top Finding 1: Best Acquisition Channels}
  \begin{columns}[T,totalwidth=\textwidth]
    \column{0.48\textwidth}
      {\color{Slate}\bfseries Question: Which channels generate most leads and value?}\par
      \vspace{0.2cm}
      \begin{itemize}\small
        \item Search and Meta generate the highest volume of leads.
        \item Organic and Direct channels, although lower volume, drive higher value customers (AOV \& Repeat Buying).
      \end{itemize}
    \column{0.48\textwidth}
      \figband{\linewidth}{3cm}{channel_lcr.png}
      \vspace{0.1cm}
      \callout[Forest]{\textbf{Action:} Shift 15\% budget from low-conversion Display to Search.}
  \end{columns}
\end{frame}

% ---------------------------------------------------------------------
% SLIDE 5
% ---------------------------------------------------------------------
\begin{frame}{Top Finding 2: Campaign Efficiency vs Quality}
  \begin{columns}[T,totalwidth=\textwidth]
    \column{0.48\textwidth}
      {\color{Slate}\bfseries Question: Which campaigns are surface-efficient?}\par
      \vspace{0.2cm}
      \begin{itemize}\small
        \item Awareness campaigns generate cheap clicks but very low revenue/ROAS.
        \item Targeted Lead Gen campaigns demonstrate much higher ROAS.
      \end{itemize}
    \column{0.48\textwidth}
      \figband{\linewidth}{3.5cm}{efficiency_frontier.png}
  \end{columns}
  \vfill
  \callout[Slate]{\textbf{Action:} Redesign Awareness campaigns to include stronger call-to-actions, or pause them in favor of lead-gen.}
\end{frame}

% ---------------------------------------------------------------------
% SLIDE 6
% ---------------------------------------------------------------------
\begin{frame}{Top Finding 3: Regional \& Device Patterns}
  \begin{columns}[T,totalwidth=\textwidth]
    \column{0.48\textwidth}
      \card{Slate}{Regional Performance}{
        \begin{itemize}\small
          \item South East region shows highest conversion and revenue.
          \item Other regions underperform relative to spend.
        \end{itemize}
      }
    \column{0.48\textwidth}
      \card{Teal}{Device Discrepancy}{
        \begin{itemize}\small
          \item Mobile users account for 60\% of sessions but only 40\% of revenue.
          \item Indicates a high bounce rate or checkout friction on mobile.
        \end{itemize}
      }
  \end{columns}
  \vfill
  \callout[Amber]{\textbf{Action:} Optimize mobile checkout experience immediately.}
\end{frame}

% ---------------------------------------------------------------------
% SLIDE 7
% ---------------------------------------------------------------------
\begin{frame}{Top Finding 4: Discounting Effectiveness}
  \begin{columns}[T,totalwidth=\textwidth]
    \column{0.48\textwidth}
      {\color{Slate}\bfseries Question: Does discounting justify margin sacrifice?}\par
      \vspace{0.2cm}
      \begin{itemize}\small
        \item 10-15\% discounts significantly lift LCR for New Customers.
        \item Discounts > 20\% do not improve LCR enough to justify the margin loss.
      \end{itemize}
    \column{0.48\textwidth}
      \card{Forest}{Recommendation}{
        \begin{itemize}\small
          \item Cap discounts at 15\% for new leads.
          \item Stop discounting for 'Champions' segment, who are price-insensitive.
        \end{itemize}
      }
  \end{columns}
\end{frame}

% ---------------------------------------------------------------------
% SLIDE 8

\begin{frame}{Top Finding 5: Customer Segments \& Value}
  \begin{columns}[T,totalwidth=\textwidth]
    \column{0.48\textwidth}
      \card{Slate}{K-Means Clustering}{Identified 4 distinct segments based on transaction and behavioral data.}
      \vspace{0.2cm}
      \begin{itemize}\small
        \item 'Champions' deserve VIP treatment, early access, no deep discounts.
        \item 'At Risk' segment needs targeted re-engagement campaigns.
      \end{itemize}
    \column{0.48\textwidth}
      \card{Teal}{Behavioral Signals}{
        \begin{itemize}\small
          \item High correlation between add-to-cart events and final purchase.
          \item Low correlation with pages-viewed for 'At Risk' segments.
        \end{itemize}
      }
  \end{columns}
\end{frame}
